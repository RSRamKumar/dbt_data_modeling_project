-- Use an admin role
USE ROLE ACCOUNTADMIN;

-- Create the `transform` role
CREATE ROLE IF NOT EXISTS TRANSFORM_ROLE;
GRANT ROLE TRANSFORM_ROLE TO ROLE ACCOUNTADMIN;

-- Create the default warehouse if necessary
CREATE WAREHOUSE IF NOT EXISTS COMPUTE_WH;
GRANT OPERATE ON WAREHOUSE COMPUTE_WH TO ROLE TRANSFORM_ROLE;

-- Create the `ram` user and assign to role
CREATE USER IF NOT EXISTS ram
  PASSWORD='******'
  LOGIN_NAME='ram'
  MUST_CHANGE_PASSWORD=FALSE
  DEFAULT_WAREHOUSE='COMPUTE_WH'
  DEFAULT_ROLE=TRANSFORM_ROLE
  DEFAULT_NAMESPACE='JAFFLE.RAW_SCHEMA'
  COMMENT='DBT user used for data transformation';
ALTER USER ram SET TYPE = LEGACY_SERVICE;
GRANT ROLE TRANSFORM_ROLE to USER ram;

-- Create our database and schemas
CREATE DATABASE IF NOT EXISTS JAFFLE;
CREATE SCHEMA IF NOT EXISTS JAFFLE.RAW_SCHEMA;

-- Set up permissions to role `transform`
GRANT ALL ON WAREHOUSE COMPUTE_WH TO ROLE TRANSFORM_ROLE; 
GRANT ALL ON DATABASE JAFFLE to ROLE TRANSFORM_ROLE;
GRANT ALL ON ALL SCHEMAS IN DATABASE JAFFLE to ROLE TRANSFORM_ROLE;
GRANT ALL ON FUTURE SCHEMAS IN DATABASE JAFFLE to ROLE TRANSFORM_ROLE;
GRANT ALL ON ALL TABLES IN SCHEMA JAFFLE.RAW_SCHEMA to ROLE TRANSFORM_ROLE;
GRANT ALL ON FUTURE TABLES IN SCHEMA JAFFLE.RAW_SCHEMA to ROLE TRANSFORM_ROLE;
